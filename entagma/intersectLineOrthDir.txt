vector dir,newPos;
float rand,lenNew;
//ortho dir
rand = floor(fit01(rand(@ptnum+1),0,2));
rand = fit(rand,0,1,-1,1);
if (rand(@ptnum+3)>=0.5)
    dir = set(rand,0,0);
else 
    dir = set(0,0,rand);
lenNew = 1;
newPos = @P+dir*lenNew;
int newPt = addpoint(0,newPos);
addprim(0,"polyline",@ptnum,newPt);
@N=rand;
